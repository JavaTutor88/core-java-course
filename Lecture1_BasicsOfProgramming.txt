package com.aspl.introduction;

class BasicsOfProgramming {

	public static void main(String[] args) {

		int x = 10 * 2;
//	int is a datatype
//	x is a variable
//	= is assignment operator
//	10 and 2 are operands/literals/constant values
//	int x = 10 * 2 is a statement that ends with semicolon
//	Similarly you can use / % + - operators
//	/ * % have higher precedence than + - operators
		System.out.println(8 + 10 / 2);
		System.out.println(6 / 2 * 3);
		System.out.println(8 % 2);
		System.out.println(25 % 4);
		System.out.println(4 % 2 * 20);
		System.out.println((6/2)%2);
		System.out.println(5.2/3);
		System.out.println("Thats it");
		
		
	}
//	Case sensitiveness, whitespace effects while printing in sysout
//	Escape character \
//	\n \t 

}
